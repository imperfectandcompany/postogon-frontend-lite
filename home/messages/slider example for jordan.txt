<!--  this library (splide.js) allows us to easily make mobile gesture slides. -->
<script src="https://cdn.jsdelivr.net/npm/@splidejs/splide@latest/dist/js/splide.min.js"></script>
<!-- reading the documentation, it says that we need to include their stylesheet as well -->
<link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/@splidejs/splide@latest/dist/css/splide.min.css">
<script>
    document.addEventListener( 'DOMContentLoaded', function () {
var elms = document.getElementsByClassName( 'splide' );
for ( var i = 0, len = elms.length; i < len; i++ ) {
    new Splide( elms[ i ], {
    arrows: false,
} ).mount();
}
    } );

</script>



<div class="splide">
    <div class="splide__track">
        <ul class="splide__list">
            <li class="splide__slide">
            <!-- Put the stuff you want to initially slide here -->
            </li>
            <li class="splide__slide">
            <!-- Put the stuff you want it to slide to here. -->
            </li>
        </ul>
    </div>
</div>
